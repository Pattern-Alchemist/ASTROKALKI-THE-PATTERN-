1. Automated Keyword Rank Tracking (Python + Google Search Console API)
# This script fetches keyword ranking data from Google Search Console API
# and saves it to a CSV for tracking keyword performance over time.

from googleapiclient.discovery import build
from oauth2client.service_account import ServiceAccountCredentials
import csv
import datetime

SCOPES = ['https://www.googleapis.com/auth/webmasters.readonly']
KEY_FILE_LOCATION = 'path_to_your_service_account_credentials.json'
SITE_URL = 'https://www.astrokalki.com'

def initialize_search_console():
    credentials = ServiceAccountCredentials.from_json_keyfile_name(KEY_FILE_LOCATION, SCOPES)
    webmasters_service = build('searchconsole', 'v1', credentials=credentials)
    return webmasters_service

def fetch_keyword_data(webmasters_service):
    today = datetime.date.today()
    start_date = (today - datetime.timedelta(days=7)).isoformat()
    end_date = today.isoformat()

    request = {
        'startDate': start_date,
        'endDate': end_date,
        'dimensions': ['query'],
        'rowLimit': 1000
    }
    response = webmasters_service.searchanalytics().query(siteUrl=SITE_URL, body=request).execute()
    return response.get('rows', [])

def save_to_csv(rows):
    with open('keyword_ranking.csv', mode='a', newline='') as file:
        writer = csv.writer(file)
        for row in rows:
            writer.writerow([row['keys'][0], row['clicks'], row['impressions'], row['ctr'], row['position']])

def main():
    service = initialize_search_console()
    rows = fetch_keyword_data(service)
    if rows:
        save_to_csv(rows)
    else:
        print('No keyword data found.')

if __name__ == '__main__':
    main()

// Send personalized backlink outreach emails from a Google Sheet list

function sendBacklinkEmails() {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Outreach');
  var startRow = 2;  
  var numRows = sheet.getLastRow() - 1;
  var dataRange = sheet.getRange(startRow, 1, numRows, 3);
  var data = dataRange.getValues();

  data.forEach(function(row, i) {
    var emailSent = row[2]; // Column C: Email Sent status
    if (emailSent !== 'SENT') {
      var emailAddress = row[0]; // Column A: Email
      var name = row[1]; // Column B: Contact Name
      var subject = 'Collaborate with Astrokalki for Quality Astrology Content';
      var message = 'Hi ' + name + ',\n\n' +
                    'I’m reaching out from Astrokalki.com, where we offer personalized astrology insights. ' +
                    'I’d love to explore a guest post or backlink collaboration to benefit both our audiences.\n\n' +
                    'Looking forward to your thoughts!\n\nBest,\n[Your Name]';

      MailApp.sendEmail(emailAddress, subject, message);

      sheet.getRange(startRow + i, 3).setValue('SENT');

To connect your Hostinger domain to your Vercel project, follow these steps:

1. Add Your Domain to Vercel:

Access Vercel Dashboard: Log in to your Vercel account and navigate to the project you wish to connect.

Add Domain: Go to the project's settings, select the "Domains" section, and click on "Add." Enter your Hostinger domain name and follow the prompts. Vercel will provide DNS records (typically an A record and a CNAME record) that you'll need to configure in Hostinger.

2. Configure DNS Settings in Hostinger:

Log in to Hostinger: Access your Hostinger account and go to the "DNS Zone Editor" for your domain.

Update DNS Records:

A Record: Delete any existing A records for the root domain (@) and add a new A record with the IP address provided by Vercel.

CNAME Record: Delete any existing CNAME records for "www" and add a new CNAME record pointing to "cname.vercel-dns.com."

Save Changes: After updating the DNS records, save the changes. DNS propagation may take up to 24 hours, but changes often take effect within a few hours.

3. Verify Domain Connection in Vercel:

Check Status: Return to your Vercel dashboard and verify that the domain status shows as "Connected."
