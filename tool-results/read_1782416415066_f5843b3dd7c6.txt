    56‚Üífrom googleapiclient.discovery import build
    57‚Üífrom oauth2client.service_account import ServiceAccountCredentials
    58‚Üíimport csv
    59‚Üíimport datetime
    60‚Üí
    61‚ÜíSCOPES = ['https://www.googleapis.com/auth/webmasters.readonly']
    62‚ÜíKEY_FILE_LOCATION = 'path_to_your_service_account_credentials.json'
    63‚ÜíSITE_URL = 'https://www.astrokalki.com'
    64‚Üí
    65‚Üídef initialize_search_console():
    66‚Üí    credentials = ServiceAccountCredentials.from_json_keyfile_name(KEY_FILE_LOCATION, SCOPES)
    67‚Üí    webmasters_service = build('searchconsole', 'v1', credentials=credentials)
    68‚Üí    return webmasters_service
    69‚Üí
    70‚Üídef fetch_keyword_data(webmasters_service):
    71‚Üí    today = datetime.date.today()
    72‚Üí    start_date = (today - datetime.timedelta(days=7)).isoformat()
    73‚Üí    end_date = today.isoformat()
    74‚Üí
    75‚Üí    request = {
    76‚Üí        'startDate': start_date,
    77‚Üí        'endDate': end_date,
    78‚Üí        'dimensions': ['query'],
    79‚Üí        'rowLimit': 1000
    80‚Üí    }
    81‚Üí    response = webmasters_service.searchanalytics().query(siteUrl=SITE_URL, body=request).execute()
    82‚Üí    return response.get('rows', [])
